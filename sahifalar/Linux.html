Terminal bilan ishlash

pwd - turgan joyimiz ko'rsatadi
ls - turgan joyimizni ichidagi narsalarni ko'rsatadi. ls -l - ketma ketlikda ma'lumotlar bilan ko'rsatadi
cd - papkani o'zgartirish uchun. Yani boshqa faylga kirish yoki chiqish uchun
mkdir - papka (folder) yaratish uchun
touch - yangi fayl yaratish uchun. Masalan script.js, index.html va hkz
cat - faylni ichidagini o'qish uchun. M.: cat index.html
cp - fayldan nusxa oladi. cp fayl nomi manzil . M.: cp doc.txt /home/nozimbek/testLinuxTerminal/newFile
mv - fayllarni qirqib oladi. mv fayl nomi manzil . M.: mv index.html /home/nozimbek/testLinuxTerminal/newFile
less - ma'lumotni o'qishimiz uchun tepaga va pastga yurishimizga kerak bo'ladi. M.: less index.html. Qo'shimcha ma'lumot uchun
more - pastga qanchadir qismini tushiradi ya'ni %da. M.: more index.html
head - boshidagi 10 qatorni ko'rsatib beradi. M.: head index.html
tail - ohiridan 10 qatorni ko'rsatadi. M.: tail index.html
nano - fayl ma'lumotlarini o'zgartirish ya'ni update
sudo -s - root maqomini olish. Ya'ni keyinchalik biror amal bajarilganda parol so'ramidi
cat /etc/passwd - ro'yhatdan o'tgan userlar ro'yhatini ko'rsatadi
cat /etc/shadow - ro'yhatdan o'tgan userlar parolini ko'rsatadi. Parolni shifrlarngan holatda ko'rsatadi havfsizlik uchun
useradd - user qo'shadi. useradd username. M.: useradd user1
useradd -m username - fayllar ustida ishlay oladigan user yaratish. M.: useradd -m yangiFoydalanuvchi
passwd - passwd username. userga parol qo'yadi. M.: passwd user1
userdel - userdel username. userni o'chiradi. M.: userdel user1
cd /etc/skel - bu yerga qo'shilgan narsalar yangi yaratilgan userlarda ham paydo bo'ladi

usermod

usermod -aG sudo username. Foydalanuvchiga sudo huquqi berish. sudo - hammanarsadan foydalanaoladi
usermod -G groupname username - sudo huquqini olib qo'yish. M.: usermod -G user1 user1. Odatda o'zini nomidagi guruhda bo'ladi

id username - usernamehaqida ma'lumot beradi qanday huquqlari borligi haqida

userlar huquqlari

belgilar

r - read. 2^2 = 4
w - write. 2^1 = 2
x - exist (mavjud bo'lmoq). 2^0 = 1
- - mavjud emas. - = 0

Birinchi uchtalik user uchun
Ikkinchi uchtail group uchun
Uchinchi uchtalik boshqalar uchun
M.: -rw-r--r--

drwxr-xr-x (rwx = 4+2+1 = 7, r-x = 4+0+1 = 5, r-x = 4+0+1 = 5) = 755

chmod - foydalanuvchilar va fayllarga huquq beradi. M.: chmod 756 fayl.txt. 756 = rwxr-xrw-

echo

echo - o'zidan keyingi ma'lumotni qaytaradi. M.: echo "Hello World" (Terminalda Hello World ni qaytaradi)
echo 'file1 file2 file3' | xargs mkdir - file1, file2 va file3 folderlarini yaratib beradi
echo 'file1 file2 file3' | xargs -t mkdir - file1, file2 va file3 folderlarini yaratib beradi va qanday folderlar yaratilgan bo'lsa shularni ko'rsatadi
echo 'file1 file2 file3' | xargs rmdir - file1, file2 va file3 folderlarini o'chiradi
echo 'file1 file2 file3' | xargs -t rmdir - file1, file2 va file3 folderlarini o'chiradi va qanday folderlar o'chirilgan bo'lsa shularni ko'rsatadi

awk

awk - faylni ichidan kerakli ma'lumotlarni filtr qilib kerakli ma'lumotlarni qaytaradi. M.: awk '{print $2}' doc.txt ($ - ustunni bildiradi ya'ni 2 ustundagi ma'lumotlarni qaytaradi)
print - ko'rsatishni bildiradi

awk '/Nozimbek/ {print}' doc.txt - Nozimbek degan ma'lumoti bor qatorni ko'rsatadi
awk 'END {print NR} doc.txt ma'lumotlar necha qatorligini qaytaradi

rmdir - folderni o'chiradi
grep - kerakli ma'lumotni o'zini qatoridagi boshqa ma'lumotlar bilan qaytaradi agar bo'lsa. M.: cat doc.txt | grep Nozimbek (Nozimbek degan ma'lumoti bor qatorni bildiradi va qidirilgan ma'lumotni qizil rangda ko'rsatadi)
